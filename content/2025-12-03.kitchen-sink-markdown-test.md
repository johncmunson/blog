## Syntax Highlighting

Shiki powers fenced code blocks. You get themed output in light and dark mode without any extra work.

```ts
interface Feature {
  name: string;
  enabled: boolean;
}

const features: Feature[] = [
  { name: "markdown", enabled: true },
  { name: "gfm", enabled: true },
];

export function isEnabled(name: string) {
  return features.some((feature) => feature.name === name && feature.enabled);
}
```
